import { notificationQueue } from "@shared/schema";
import { eq } from "drizzle-orm";

async function main() {
  console.log("üîç Fetching notification...");
  const [notification] = await db
    .select()
    .from(notificationQueue)
    .where(eq(notificationQueue.id, 8));

  if (!notification) {
    console.error("‚ùå Notification not found.");
    return;
  }

  if (notification.status !== "pending") {
    console.warn("‚ö†Ô∏è Notification already processed or not pending.");
    return;
  }

  const payload = notification.payload as {
    recipientEmail: string;
    subject: string;
    context?: { body?: string };
  };

  const to = payload.recipientEmail;
  const subject = payload.subject;
  const body = payload.context?.body || "[No message body]";

  try {
    await sendEmail(to, subject, body);

    await db.update(notificationQueue)
      .set({
        status: "sent",
        processAttempts: (notification.processAttempts || 0) + 1,
        lastAttemptAt: new Date()
      })
      .where(eq(notificationQueue.id, 8));

    console.log("‚úÖ Email sent and notification updated.");
  } catch (error) {
    if (error instanceof Error) {
      console.error("‚ùå Error sending email:", error.message);

      await db.update(notificationQueue)
        .set({
          status: "error",
          error: error.message,
          lastAttemptAt: new Date(),
          processAttempts: (notification.processAttempts || 0) + 1
        })
        .where(eq(notificationQueue.id, 8));
    } else {
      console.error("‚ùå Unknown error:", error);
    }
  }
}

main();
